      function build_srcdoc(mode, code) {
        if (mode === "html") return build_html_srcdoc(code);
        if (mode === "javascript") return build_javascript_srcdoc(code);
        if (mode === "python") return build_python_srcdoc(code);
        throw new Error("unsupported mode: " + mode);
      }

      function clear_console() {
        set_console_text("");
      }

      function run_code() {
        save_current_code();
        clear_console();

        const mode = state.current_mode;
        const code = elements.code_input.value;

        try {
          const srcdoc = build_srcdoc(mode, code);
          elements.preview_frame.srcdoc = srcdoc;

          if (mode === "html") {
            set_status("HTML を描画しました。", "ok");
          } else if (mode === "javascript") {
            set_status("JavaScript を実行しました。", "ok");
            append_console_line("[status] JavaScript iframe created");
          } else if (mode === "python") {
            set_status("Python を実行しています。初回は Pyodide 読み込みに少し時間がかかります。", "warn");
            append_console_line("[status] Python iframe created");
          }
        } catch (error) {
          set_status("実行失敗: " + (error.message || String(error)), "error");
          append_console_line("[error] " + (error.stack || error.message || String(error)));
        }
      }

      function reload_preview() {
        const blank = 'blank';
        elements.preview_frame.srcdoc = blank;
        set_status("プレビューを再読込しました。", "ok");
        append_console_line("[status] preview reloaded");
      }

      async function copy_code() {
        try {
          await navigator.clipboard.writeText(elements.code_input.value);
          set_status("コードをコピーしました。", "ok");
        } catch (error) {
          set_status("コピー失敗: " + (error.message || String(error)), "error");
        }
      }

      function confirm_reset() {
        return window.confirm("現在モードのコードをサンプルに戻します。よろしいですか？");
      }

      function confirm_clear() {
        return window.confirm("現在モードのコードを空にします。よろしいですか？");
      }

      function reset_current_mode_code() {
        if (!confirm_reset()) {
          set_status("初期化をキャンセルしました。", "warn");
          return;
        }
        elements.code_input.value = SAMPLE_CODE[state.current_mode];
        save_current_code();
        set_status("サンプルコードに戻しました。", "ok");
      }

      function clear_current_mode_code() {
        if (!confirm_clear()) {
          set_status("全削除をキャンセルしました。", "warn");
          return;
        }
        elements.code_input.value = "";
        save_current_code();
        set_status("コードを空にしました。", "ok");
      }

      function load_sample() {
        elements.code_input.value = SAMPLE_CODE[state.current_mode];
        save_current_code();
        set_status("サンプルコードを読み込みました。", "ok");
      }

      function handle_message(event) {
        const data = event.data;
        if (!data || data.source !== "single_file_ide_preview") {
          return;
        }

        if (data.kind === "log") {
          append_console_line("[log] " + data.message);
          return;
        }

        if (data.kind === "warn") {
          append_console_line("[warn] " + data.message);
          return;
        }

        if (data.kind === "error") {
          append_console_line("[error] " + data.message);
          set_status("実行中にエラーが発生しました。", "error");
          return;
        }

        if (data.kind === "status") {
          append_console_line("[status] " + data.message);
        }
      }

      function bind_events() {
        elements.language_select.addEventListener("change", function () {
          switch_mode(elements.language_select.value);
        });

        elements.code_input.addEventListener("input", function () {
          save_current_code();
        });

        elements.sample_button.addEventListener("click", load_sample);
        elements.run_button.addEventListener("click", run_code);
        elements.reload_preview_button.addEventListener("click", reload_preview);
        elements.copy_button.addEventListener("click", copy_code);
        elements.reset_button.addEventListener("click", reset_current_mode_code);
        elements.clear_button.addEventListener("click", clear_current_mode_code);

        window.addEventListener("message", handle_message);
      }

      function start() {
        const saved_mode = localStorage.getItem(STORAGE_KEY_MODE);
        state.current_mode = saved_mode || "html";
        if (!SAMPLE_CODE[state.current_mode]) {
          state.current_mode = "html";
        }

        elements.language_select.value = state.current_mode;
        elements.code_input.value = load_code_for_mode(state.current_mode);
        update_editor_meta();
        bind_events();
        set_status("準備完了。モードを選んで実行してください。", "ok");
        reload_preview();
      }

      start();
